Generador de número aleatorio


function getRandomNumber(min, max) {
  return Math.floor(Math.random() * (max - min + 1)) + min;
}


Comprobar si un array está vacío


const isEmptyArray = arr => Array.isArray(arr) && arr.length === 0;


Comprobar si un objeto está vacío


const isEmptyObject = obj => Object.keys(obj) === 0;


Convertir Camel Case a Snake Case


const camelToSnake = str => str.replace(/([A-Z])/g, "_$1").tolowerCase();


Convertir a mayúscula la primera letra de cada palabra


const capitalizeWords = str => str.replace(/\b\w\g, char => char.toUpperCase());


Convertir texto a Title Case


const toTitleCase = str => str.toLowerCase().split(' ')
      .map(word => word.charAt(0).toUpperCase() + word.slice(1))
      .join(' ');

Barajar un array de objetos


const sortByKey = (array, key) => array.sort((a, b) => (a[key] > b[key]) ? 1 : -1);

Generar UUID


const generateUUID = () => 
      'xxxxxxxx-xxx-4xxx-yxxx-xxxxxxxx'
      .replace(/[xy]/g, c => (Math.random() * 16 | 0).toString(16));


Seleccionar un número aleatorio de un array


const getRandomElement = arr => arr[Math.floor(Math.random() * arr.length)];


Convertir de Celsius a Fahrenheit


const celsiusToFahrenheit = celsius => (celsius * 9/5 + 32);


Convertir un array a un objeto


const arrayToObject = (arr, key) => arr.reduce((obj, item) => {
      obj[item[key]] = item; return obj; }, {});
